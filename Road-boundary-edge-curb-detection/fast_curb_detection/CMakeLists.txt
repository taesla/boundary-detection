project(fast_curb_detection)


find_package(catkin REQUIRED COMPONENTS
  pcl_conversions
  pcl_ros
  roscpp
  rospy
  std_msgs
  grid_map_ros
  grid_map_cv
  grid_map_msgs
  autoware_msgs
  #velodyne_pointcloud
)

catkin_package(CATKIN_DEPENDS)

find_package(PCL REQUIRED)

	include
